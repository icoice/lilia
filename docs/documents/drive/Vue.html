drive.Vue.store

step 1

export default (store) => {
  const { plugin } = store;

  store.state({
    list: null,
    total: 0,
  });

  store.action('get', ({ publish }, params = {}) => {
    publish({
      list: [1, 2, 3, 4],
      total: 3
    });
  });
}

step 2

import vue from 'vue';
import { drive } from 'vue-moo';
import demo from './map/demo';

export default drive.Vue.store(vue, (Store) => {
  Store.use('plugin', {});
  demo(new Store('demo'));
});
